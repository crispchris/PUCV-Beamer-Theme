% Showcase Slide: Pseudocódigo de Algoritmo
\pucvframe{PSEUDOCÓDIGO DE ALGORITMOS}{
Tipografía y formateo de algoritmos mediante el uso de los paquetes \texttt{algorithm} y \texttt{algpseudocode}:

\vspace{10bp}

\begin{algorithm}[H]
    \centering
    \fontsize{14bp}{17bp}\selectfont
    \color{PUCVText}
    \begin{algorithmic}[1]
        \Procedure{GradientDescent}{$\theta, \alpha, N$}
            \For{$t \leftarrow 1$ to $N$}
                \State $g_t \leftarrow \nabla_\theta L(\theta)$
                \State $\theta \leftarrow \theta - \alpha g_t$
            \EndFor
            \State \textbf{return} $\theta$
        \EndProcedure
    \end{algorithmic}
    \caption{Algoritmo de Descenso de Gradiente.}
    \label{alg:gd}
\end{algorithm}
}
